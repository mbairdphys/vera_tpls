132: PetscErrorCode PETSCMAP1(VecScatterEnd)(VecScatter ctx,Vec xin,Vec yin,InsertMode addv,ScatterMode mode)
133: {
134:   VecScatter_MPI_General *to,*from;

139:   MPI_Request            *rwaits,*swaits;
